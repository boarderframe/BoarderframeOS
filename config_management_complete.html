📝 Usage Example

from core.config import RedisConfigCache, ConfigCacheManager

# Initialize cache
cache = RedisConfigCache(
    redis_url="redis://localhost:6379",
    postgres_dsn="postgresql://...",
    ttl_seconds=300,
    refresh_threshold=0.8
)
await cache.initialize()

# Create cache manager
manager = ConfigCacheManager(cache)
await manager.start()

# Get agent config (< 5ms for cache hits)
config = await manager.get_config("Solomon")

# Invalidate department
await manager.invalidate_department("Executive")

# Get cache health
health = await manager.get_cache_health()
